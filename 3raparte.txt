```rust
}

    pub events: Vec<EventBinding>,
    pub enabled: bool,

#[serde(untagged)]

    String(String),
    Integer(i64),
    Float(f64),
    Boolean(bool),
    Vec2([f32; 2]),
    Vec3([f32; 3]),
    Vec4([f32; 4]),
    Color([u8; 4]),
    AssetPath(String),
    NodeRef(Uuid),
}
```

```rust
use super::*;

pub struct OscillateBehavior {
    pub axis: String,      // "x", "y", "z"
    pub speed: f32,
    pub amplitude: f32,
    pub offset: f32,
    pub relative: bool,    // true: relativo a posición inicial
}

impl Behavior for OscillateBehavior {
    fn tick(&mut self, node: &mut Node, delta: f32, time: f64) {
        let value = (time * self.speed as f64).sin() as f32 * self.amplitude;
        
        if let Some(transform) = node.get_component_mut::<TransformComponent>() {
            match self.axis.as_str() {
                "x" => transform.position[0] = value,
                "y" => transform.position[1] = value,
                "z" => transform.position[2] = value,
                _ => {}
            }
        }
    }
}
```

```rust
// scripting/behaviors/patrol_ai.rs
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct PatrolAI {
    pub waypoints: Vec<[f32; 3]>,
    pub speed: f32,
    pub wait_time: f32,
    pub current_target: usize,
    pub wait_timer: f32,
    pub state: PatrolState,
}

#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
pub enum PatrolState {
    Moving,
    Waiting,
}

impl Behavior for PatrolAI {
    fn tick(&mut self, node: &mut Node, delta: f32, _time: f64) {
        if self.waypoints.is_empty() {
            return;
        }
        
        match self.state {
            PatrolState::Moving => {
                let target = self.waypoints[self.current_target];
                let mut transform = node.get_component_mut::<TransformComponent>().unwrap();
                
                // Moverse hacia el target
                let dir = [
                    target[0] - transform.position[0],
                    target[1] - transform.position[1],
                    target[2] - transform.position[2],
                ];
                
                let distance = (dir[0]*dir[0] + dir[1]*dir[1] + dir[2]*dir[2]).sqrt();
                
                if distance < 0.1 {
                    // Llegó al waypoint
                    self.state = PatrolState::Waiting;
                    self.wait_timer = self.wait_time;
                    
                    // Siguiente waypoint
                    self.current_target = (self.current_target + 1) % self.waypoints.len();
                } else {
                    // Normalizar y mover
                    let scale = (self.speed * delta) / distance;
                    transform.position[0] += dir[0] * scale;
                    transform.position[1] += dir[1] * scale;
                    transform.position[2] += dir[2] * scale;
                }
            }
            
            PatrolState::Waiting => {
                self.wait_timer -= delta;
                if self.wait_timer <= 0.0 {
                    self.state = PatrolState::Moving;
                }
            }
        }
    }
}
```

```rust
// scripting/ecs/world.rs
use std::collections::{HashMap, HashSet};

pub struct EcsWorld {
    entities: HashMap<Uuid, Entity>,
    systems: Vec<Box<dyn System>>,
    behavior_instances: HashMap<Uuid, Box<dyn Behavior>>,
    events: Vec<Event>,
}

pub struct Entity {
    pub id: Uuid,
    pub node: Node,
    pub active: bool,
}

impl EcsWorld {
    pub fn new() -> Self {
        EcsWorld {
            entities: HashMap::new(),
            systems: Vec::new(),
            behavior_instances: HashMap::new(),
            events: Vec::new(),
        }
    }
    
    pub fn load_scene(&mut self, scene: Scene) {
        self.entities.clear();
        self.behavior_instances.clear();
        
        // Crear entidades para cada nodo
        for node in scene.nodes {
            let id = node.id;
            
            // Instanciar comportamientos
            for component in &node.components {
                if let Component::Behavior(behavior_data) = component {
                    if let Some(behavior) = self.instantiate_behavior(behavior_data) {
                        self.behavior_instances.insert(id, behavior);
                    }
                }
            }
            
            self.entities.insert(id, Entity {
                id,
                node,
                active: true,
            });
        }
        
        // Establecer relaciones padre-hijo
        self.build_hierarchy();
    }
    
    pub fn update(&mut self, delta: f32) {
        let time = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .unwrap()
            .as_secs_f64();
        
        // Actualizar todos los comportamientos
        for (entity_id, behavior) in &mut self.behavior_instances {
            if let Some(entity) = self.entities.get_mut(entity_id) {
                if entity.active {
                    behavior.tick(&mut entity.node, delta, time);
                }
            }
        }
        
        // Ejecutar sistemas
        for system in &mut self.systems {
            system.run(&mut self.entities, delta);
        }
        
        // Procesar eventos
        self.process_events();
    }
    
    fn instantiate_behavior(&self, data: &BehaviorComponent) -> Option<Box<dyn Behavior>> {
        match data.behavior_id.as_str() {
            "oscillate" => {
                let props = &data.properties;
                Some(Box::new(OscillateBehavior {
                    axis: props.get("axis").and_then(|v| v.as_string()).unwrap_or("y".to_string()),
                    speed: props.get("speed").and_then(|v| v.as_float()).unwrap_or(1.0) as f32,
                    amplitude: props.get("amplitude").and_then(|v| v.as_float()).unwrap_or(1.0) as f32,
                    offset: 0.0,
                    relative: true,
                }))
            }
            "patrol_ai" => {
                let props = &data.properties;
                Some(Box::new(PatrolAI {
                    waypoints: props.get("waypoints").and_then(|v| v.as_vec3_array()).unwrap_or_default(),
                    speed: props.get("speed").and_then(|v| v.as_float()).unwrap_or(2.0) as f32,
                    wait_time: props.get("wait_time").and_then(|v| v.as_float()).unwrap_or(1.0) as f32,
                    current_target: 0,
                    wait_timer: 0.0,
                    state: PatrolState::Moving,
                }))
            }
            _ => None,
        }
    }
}
```

```rust
// scripting/ecs/events.rs
#[derive(Debug, Clone)]
pub enum Event {
    Collision {
        entity_a: Uuid,
        entity_b: Uuid,
        position: [f32; 3],
    },
    Timer {
        entity: Uuid,
        timer_id: String,
    },
    Input {
        entity: Uuid,
        action: String,
        pressed: bool,
    },
    Custom {
        entity: Uuid,
        name: String,
        data: HashMap<String, PropertyValue>,
    },
}

impl EcsWorld {
    fn process_events(&mut self) {
        let events = std::mem::take(&mut self.events);
        
        for event in events {
            match event {
                Event::Collision { entity_a, entity_b, position } => {
                    // Notificar a los comportamientos de ambas entidades
                    if let Some(behavior) = self.behavior_instances.get_mut(&entity_a) {
                        behavior.on_event("on_collision", &[
                            ("other", PropertyValue::NodeRef(entity_b)),
                            ("position", PropertyValue::Vec3(position)),
                        ]);
                    }
                    
                    if let Some(behavior) = self.behavior_instances.get_mut(&entity_b) {
                        behavior.on_event("on_collision", &[
                            ("other", PropertyValue::NodeRef(entity_a)),
                            ("position", PropertyValue::Vec3(position)),
                        ]);
                    }
                }
                
                Event::Timer { entity, timer_id } => {
                    if let Some(behavior) = self.behavior_instances.get_mut(&entity) {
                        behavior.on_event("on_timer", &[
                            ("timer_id", PropertyValue::String(timer_id)),
                        ]);
                    }
                }
                
                _ => {}
            }
        }
    }
}
```

```rust
pub struct AIAssistant {

pub enum AIProvider {
    Local(LocalLLM),      // llama.cpp, etc.
    Remote(RemoteAPI),    // OpenAI, Claude, etc.
}

pub struct AssistantContext {
    current_scene: Option<Scene>,
    selected_nodes: Vec<Uuid>,
    recent_actions: Vec<String>,
}

pub struct Suggestion {
    pub id: Uuid,
    pub title: String,
    pub description: String,
    pub action: SuggestionAction,
    pub confidence: f32,
}

pub enum SuggestionAction {
    AddBehavior {
        node_id: Uuid,
        behavior_toml: String,
    },
    ModifyProperty {
        node_id: Uuid,
        property: String,
        value: PropertyValue,
    },
    GenerateScene {
        prompt: String,
        toml: String,
    },
    OptimizeTip {
        message: String,
        code: Option<String>,
    },
}

impl AIAssistant {
    pub fn new() -> Self {
        AIAssistant {
            provider: AIProvider::Local(LocalLLM::new()), // Por defecto local
            context: AssistantContext::default(),
            suggestions: Vec::new(),
            chat_history: Vec::new(),
        }
    }
    
    pub fn process_prompt(&mut self, prompt: &str) -> Result<Vec<Suggestion>> {
        // Construir contexto para la IA
        let context = self.build_context();
        
        // Enviar a la IA
        let response = self.provider.complete(&format!(
            "Contexto actual: {}\n\nUsuario: {}\n\nGenera sugerencias en JSON:",
            context, prompt
        ))?;
        
        // Parsear respuesta
        let suggestions = self.parse_response(&response)?;
        self.suggestions = suggestions;
        
        Ok(self.suggestions.clone())
    }
    
    pub fn generate_behavior(&self, description: &str) -> Result<String> {
        // Generar TOML de comportamiento desde descripción
        let prompt = format!(
            "Genera un comportamiento en TOML para un juego que: {}\n\
            Usa el formato: type = 'behavior_name', properties = {{...}}",
            description
        );
        
      self.provider.complete(&prompt)
    }
    
    fn build_context(&self) -> String {
        let mut ctx = String::new();
        
        if let Some(scene) = &self.context.current_scene {
            ctx.push_str(&format!("Escena: {}\n", scene.name));
            ctx.push_str(&format!("Nodos: {}\n", scene.nodes.len()));
        }
        
        if !self.context.selected_nodes.is_empty() {
            ctx.push_str(&format!("Nodos seleccionados: {:?}\n", self.context.selected_nodes));
        }
        
        ctx
    }
}
```

```rust
        });
        
        // Botón para generar comportamiento
        if ui.button("🎯 Generar comportamiento desde descripción").clicked() {
            // Abrir diálogo
        }
    }

        self.messages.push(ChatMessage {
            sender: Sender::User,
            text: text.to_string(),
        });

            sender: Sender::Assistant,
```

```rust
    }
    
    pub fn compress_texture(&mut self, id: Uuid, format: CompressionFormat) -> Result<()> {
        if let Some(texture) = self.textures.get_mut(&id) {
            if let Some(data) = &texture.data {
                // Comprimir textura según formato
                let compressed = match format {
                    CompressionFormat::ETC2 => compress_etc2(data, texture.width, texture.height),
                    CompressionFormat::ASTC => compress_astc(data, texture.width, texture.height, 4),
                    CompressionFormat::PNG => compress_png(data),
                }?;
                
                texture.data = Some(compressed);
                texture.format = format.into();
                texture.size_bytes = texture.data.as_ref().map(|d| d.len()).unwrap_or(0);
            }
        }
        
        Ok(())
    }
    
    fn queue_node_assets(&mut self, node: &Node) {
        for component in &node.components {
            match component {
                Component::Mesh(mesh) => {
                    self.load_queue.push(AssetLoadRequest::Mesh {
                        id: mesh.id,
                        path: mesh.path.clone(),
                        priority: Priority::Normal,
                    });
                }
                Component::Sprite(sprite) => {
                    self.load_queue.push(AssetLoadRequest::Texture {
                        id: sprite.texture_id,
                        path: sprite.path.clone(),
                        priority: Priority::Normal,
                    });
                }
                _ => {}
            }
        }
    }
    
    fn process_load_queue(&mut self) -> Result<()> {
        // Ordenar por prioridad
        self.load_queue.sort_by_key(|r| r.priority());
        
        while let Some(request) = self.load_queue.pop() {
            match request {
                AssetLoadRequest::Mesh { id, path, .. } => {
                    let data = std::fs::read(&path)?;
                    let size = data.len();
                    
                    if self.memory_budget.can_allocate(size) {
                        self.meshes.insert(id, MeshAsset {
                            id,
                            path,
                            data: Some(data),
                            loaded: true,
                            size_bytes: size,
                            lod_levels: Vec::new(),
                        });
                        
                        self.memory_budget.allocate(size);
                    }
                }
                AssetLoadRequest::Texture { id, path, .. } => {
                    // Cargar imagen y comprimir según dispositivo
                    let img = image::open(&path)?;
                    let data = img.to_rgba8().into_raw();
                    let size = data.len();
                    
                    // Aplicar compresión si es necesario
                    let (compressed_data, format) = if self.compression_settings.enabled {
                        let compressed = compress_etc2(&data, img.width(), img.height())?;
                        (Some(compressed), TextureFormat::ETC2)
                    } else {
                        (Some(data), TextureFormat::RGBA8)
                    };
                    
                    if self.memory_budget.can_allocate(size) {
                        self.textures.insert(id, TextureAsset {
                            id,
                            path,
                            data: compressed_data,
                            loaded: true,
                            width: img.width(),
                            height: img.height(),
                            format,
                            compression: CompressionType::None,
                        });
                        
                        self.memory_budget.allocate(size);
                    }
                }
            }
        }
        
        Ok(())
    }
}
```

```rust
// assets/memory_budget.rs
pub struct MemoryBudget {
    total_bytes: usize,
    used_bytes: usize,
    max_bytes: usize,
    low_memory_mode: bool,
}

impl MemoryBudget {
    pub fn new(max_mb: usize) -> Self {
        MemoryBudget {
            total_bytes: max_mb * 1024 * 1024,
            used_bytes: 0,
            max_bytes: max_mb * 1024 * 1024,
            low_memory_mode: false,
        }
    }
    
    pub fn can_allocate(&self, bytes: usize) -> bool {
        if self.low_memory_mode {
            self.used_bytes + bytes <= self.max_bytes * 80 / 100 // 80% en modo bajo
        } else {
            self.used_bytes + bytes <= self.max_bytes
        }
    }
    
    pub fn allocate(&mut self, bytes: usize) {
        self.used_bytes += bytes;
        
        if self.used_bytes > self.max_bytes * 90 / 100 {
            self.low_memory_mode = true;
        }
    }
    
    pub fn free(&mut self, bytes: usize) {
        self.used_bytes = self.used_bytes.saturating_sub(bytes);
        
        if self.used_bytes < self.max_bytes * 70 / 100 {
            self.low_memory_mode = false;
        }
    }
    
    pub fn get_usage_percent(&self) -> f32 {
        self.used_bytes as f32 / self.max_bytes as f32 * 100.0
    }
    
    pub fn suggest_compression(&self) -> CompressionSuggestion {
        if self.get_usage_percent() > 80.0 {
            CompressionSuggestion::CompressAll
        } else if self.get_usage_percent() > 60.0 {
            CompressionSuggestion::CompressTextures
        } else {
            CompressionSuggestion::None
        }
    }
}

pub enum CompressionSuggestion {
    None,
    CompressTextures,
    CompressAll,
}
```

```rust
pub struct PreviewSystem {

}

pub struct BuildCache {
    last_scene_hash: u64,
    last_apk_path: PathBuf,
}

impl PreviewSystem {
    pub fn new() -> Self {
        PreviewSystem {
            adb: ADBClient::new(),
            connected_devices: Vec::new(),
            last_build_time: std::time::Instant::now(),
            build_cache: BuildCache::new(),
        }
    }
    
    pub fn refresh_devices(&mut self) -> Result<()> {
        self.connected_devices = self.adb.list_devices()?;
        Ok(())
    }
    
    pub fn run_on_device(&mut self, device: &DeviceInfo, scene: &Scene, render_mode: RenderMode) -> Result<()> {
        // 1. Detectar si necesita rebuild
        let scene_hash = self.hash_scene(scene);
        
        if scene_hash != self.build_cache.last_scene_hash {
            // 2. Empaquetar escena para el dispositivo específico
            let apk_path = self.build_preview_apk(scene, device, render_mode)?;
            self.build_cache.last_scene_hash = scene_hash;
            self.build_cache.last_apk_path = apk_path;
        }
        
        // 3. Instalar en dispositivo
        self.adb.install(&self.build_cache.last_apk_path, &device.id)?;
        
        // 4. Ejecutar
        self.adb.run_activity(&device.id, "com.tueditor.preview/.MainActivity")?;
        
        Ok(())
    }
    
    fn build_preview_apk(&self, scene: &Scene, device: &DeviceInfo, render_mode: RenderMode) -> Result<PathBuf> {
        // Crear APK mínimo con solo los assets necesarios
        let mut builder = PreviewApkBuilder::new();
        
        // Configurar según dispositivo
        builder.set_target_abis(&device.abis);
        builder.set_render_mode(render_mode);
        
        // Aplicar optimizaciones según RAM
        if device.total_ram_mb <= 2048 {
            builder.set_quality_profile(QualityProfile::Low2GB);
        } else if device.total_ram_mb <= 4096 {
            builder.set_quality_profile(QualityProfile::Mid4GB);
        } else {
            builder.set_quality_profile(QualityProfile::High);
        }
        
        // Incluir solo assets de la escena
        builder.add_scene(scene);
        
        // Compilar APK
        let apk_path = builder.build()?;
        
        Ok(apk_path)
    }
    
    fn hash_scene(&self, scene: &Scene) -> u64 {
        use std::collections::hash_map::DefaultHasher;
        use std::hash::{Hash, Hasher};
        
        let mut hasher = DefaultHasher::new();
        scene.name.hash(&mut hasher);
        scene.nodes.len().hash(&mut hasher);
        // ... hash de otros campos relevantes
        hasher.finish()
    }
}

pub struct PreviewApkBuilder {
    target_abis: Vec<String>,
    render_mode: RenderMode,
    quality: QualityProfile,
    assets: Vec<AssetData>,
    manifest: String,
}

impl PreviewApkBuilder {
    pub fn build(&self) -> Result<PathBuf> {
        let build_dir = PathBuf::from("target/preview");
        std::fs::create_dir_all(&build_dir)?;
        
        // 1. Generar AndroidManifest.xml
        self.generate_manifest(&build_dir)?;
        
        // 2. Copiar librerías nativas según ABI
        for abi in &self.target_abis {
            self.copy_native_lib(&build_dir, abi)?;
        }
        
        // 3. Empaquetar assets (solo los necesarios)
        self.pack_assets(&build_dir)?;
        
        // 4. Compilar APK con aapt2
        self.run_aapt2(&build_dir)?;
        
        // 5. Firmar con debug key (para preview)
        self.sign_apk(&build_dir)?;
        
        Ok(build_dir.join("preview.apk"))
    }
    
    fn copy_native_lib(&self, build_dir: &Path, abi: &str) -> Result<()> {
        let lib_name = match self.render_mode {
            RenderMode::Bestia => "libwgpu_backend.so",
            RenderMode::Dinamico => "libraylib_backend.so",
        };
        
        let src = PathBuf::from(format!("target/{}/release/{}", abi, lib_name));
        let dest = build_dir.join("lib").join(abi).join(lib_name);
        
        std::fs::create_dir_all(dest.parent().unwrap())?;
        std::fs::copy(src, dest)?;
        
        Ok(())
    }
}
```

```rust
        self.scene_manager.open_scene(path)?;

        if let Some(scene) = &self.scene_manager.current_scene {

        if let Some(scene) = &self.scene_manager.current_scene {
            self.ecs_world.load_scene(scene.clone());
        }
        
        Ok(())
    }

        // Actualizar ECS (comportamientos)
        if self.mode == EditorMode::Game {
            self.ecs_world.update(delta);
        }
        
        // Actualizar renderizador
        if let Some(scene) = &self.scene_manager.current_scene {
            self.renderer.render_scene(scene);
        }
        
        // Actualizar métricas
        let metrics = self.renderer.get_metrics();
        // Mostrar en UI
    }
    
    pub fn toggle_play_mode(&mut self) {
        match self.mode {
            EditorMode::Edit => {
                // Guardar estado de edición
                self.mode = EditorMode::Game;
                self.viewport_mode = ViewportMode::Game;
            }
            EditorMode::Game => {
                // Restaurar estado de edición
                self.mode = EditorMode::Edit;
                self.viewport_mode = ViewportMode::Mode3D;
            }
        }
    }
    
    pub fn switch_render_mode(&mut self, mode: RenderMode) -> Result<()> {
        // Guardar cámara actual
        let camera = self.renderer.get_camera();
        
        // Crear nuevo renderer
        self.renderer = match mode {
            RenderMode::Bestia => Box::new(WgpuBackend::new()?),
            RenderMode::Dinamico => Box::new(RaylibBackend::new()?),
        };
        
        // Recargar escena
        if let Some(scene) = &self.scene_manager.current_scene {
            self.renderer.load_scene(scene);
            self.renderer.set_camera(&camera);
        }
        
        Ok(())
    }
}

fn create_optimal_backend(caps: &HardwareCaps) -> Box<dyn RenderBackend> {
    // Intentar wgpu primero
    if caps.vulkan_supported {
        if let Ok(wgpu) = WgpuBackend::new() {
            return Box::new(wgpu);
        }
    }
    
    // Fallback a raylib (siempre funciona)
    Box::new(RaylibBackend::new().unwrap())
}
```
