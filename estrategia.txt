5.2 Script de Lanzamiento Automatizado

```bash
#!/bin/bash
# release.sh - Script de lanzamiento automatizado

set -e

VERSION="1.0.0"
echo "🚀 Preparando lanzamiento v$VERSION"

# 1. Ejecutar tests
echo "📋 Ejecutando tests..."
cargo test --release

# 2. Compilar para todas las arquitecturas
echo "🔨 Compilando..."
./build_android.sh

# 3. Generar reporte de rendimiento
echo "📊 Generando reporte..."
cargo run --bin performance_report

# 4. Firmar APK
echo "🔏 Firmando APK..."
./sign_apk.sh "target/android-build/rust-editor-v$VERSION.apk"

# 5. Verificar APK
echo "✅ Verificando APK..."
apksigner verify "target/android-build/rust-editor-v$VERSION-signed.apk"

# 6. Generar changelog
echo "📝 Generando changelog..."
git log --pretty=format:"- %s" $(git describe --tags --abbrev=0)..HEAD > changelog.txt

# 7. Crear release en GitHub
echo "🐙 Creando release en GitHub..."
gh release create "v$VERSION" \
    "target/android-build/rust-editor-v$VERSION-signed.apk" \
    --title "Rust Editor v$VERSION" \
    --notes-file changelog.txt

echo "✅ Lanzamiento completado!"
```
