1. VISIÓN GENERAL DEL PROYECTO

1.1 Objetivo Principal

Crear un editor visual de juegos para Android (con una sola escena activa) usando Rust + wgpu (Modo Bestia) + raylib (Modo Dinámico) + egui, con scripting basado en TOML/JSON, que funcione fluidamente en dispositivos desde 2GB RAM hasta gama alta.

· Una escena activa siempre: Cerrar, guardar, abrir (nunca múltiples escenas en memoria)

· Doble backend: wgpu para máximo rendimiento, raylib para máxima compatibilidad
· Educación proactiva: Panel de métricas que enseña a optimizar
· Modo juego instantáneo: Toggle con mismo backend (sin recarga)

```rust
// core/render_backend.rs
pub trait RenderBackend: Send + Sync {
    // Inicialización
    fn new() -> Result<Self, RenderError> where Self: Sized;
    
    // Ciclo de renderizado
    fn begin_frame(&mut self);
    fn render_scene(&mut self, scene: &Scene);
    fn end_frame(&mut self);
    
    // Gestión de recursos
    fn load_mesh(&mut self, mesh_data: &MeshData) -> Result<MeshHandle, RenderError>;
    fn load_texture(&mut self, texture_data: &TextureData) -> Result<TextureHandle, RenderError>;
    fn unload_mesh(&mut self, handle: MeshHandle);
    fn unload_texture(&mut self, handle: TextureHandle);
    
    // Cámara
    fn set_camera(&mut self, camera: &Camera);
    fn get_camera(&self) -> Camera;
    
    // Métricas (para el panel)
    fn get_metrics(&self) -> PerformanceMetrics;
    
    // Validación
    fn validate_frame(&self) -> bool;
    fn get_backend_type(&self) -> BackendType;
}

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum BackendType {
    Wgpu,   // Modo Bestia
    Raylib, // Modo Dinámico
}

#[derive(Debug, Clone)]
pub struct PerformanceMetrics {
    pub ram_mb: f32,
    pub cpu_percent: f32,
    pub gpu_percent: f32,
    pub fps: u32,
    pub draw_calls: u32,
    pub polygons: u32,
    pub frame_time_ms: f32,
}
```

3. SISTEMA DE UNA ESCENA ACTIVA

3.1 Scene Manager

```rust
// core/scene_manager.rs
pub struct SceneManager {
    current_scene: Option<Scene>,
    scene_path: Option<PathBuf>,
    modified: bool,
    undo_stack: Vec<SceneSnapshot>,
    redo_stack: Vec<SceneSnapshot>,
}

pub struct Scene {
    pub id: Uuid,
    pub name: String,
    pub metadata: SceneMetadata,
    pub nodes: Vec<Node>,
    pub animations: Vec<Animation>,
    pub physics: PhysicsSettings,
    pub camera: Camera,
    pub assets: SceneAssets,
}

pub struct SceneSnapshot {
    scene: Scene,
    timestamp: std::time::Instant,
    description: String,
}

impl SceneManager {
    pub fn new() -> Self {
        SceneManager {
            current_scene: None,
            scene_path: None,
            modified: false,
            undo_stack: Vec::with_capacity(50), // Máximo 50 deshacer
            redo_stack: Vec::new(),
        }
    }
    
    pub fn open_scene(&mut self, path: PathBuf) -> Result<(), SceneError> {
        // 1. Preguntar guardar si hay modificaciones
        if self.modified {
            self.prompt_save()?;
        }
        
        // 2. Liberar escena anterior (¡CRÍTICO!)
        if let Some(old_scene) = self.current_scene.take() {
            self.unload_scene_assets(old_scene);
        }
        
        // 3. Cargar nueva escena desde TOML
        let content = std::fs::read_to_string(&path)?;
        let scene: Scene = toml::from_str(&content)?;
        
        // 4. Validar recursos
        self.validate_scene_assets(&scene)?;
        
        // 5. Asignar como activa
        self.current_scene = Some(scene);
        self.scene_path = Some(path);
        self.modified = false;
        
        // 6. Resetear stacks de undo/redo
        self.undo_stack.clear();
        self.redo_stack.clear();
        
        Ok(())
    }
    
    pub fn close_scene(&mut self) {
        // Liberar TODO explícitamente
        if let Some(scene) = self.current_scene.take() {
            self.unload_scene_assets(scene);
        }
        
        self.scene_path = None;
        self.modified = false;
        self.undo_stack.clear();
        self.redo_stack.clear();
    }
    
    fn unload_scene_assets(&self, scene: Scene) {
        // Notificar al renderizador que libere recursos
        // Importante: Rust drop libera memoria, pero necesitamos
        // liberar recursos gráficos explícitamente
        for mesh in scene.assets.meshes {
            renderer.unload_mesh(mesh.handle);
        }
        for texture in scene.assets.textures {
            renderer.unload_texture(texture.handle);
        }
    }
}
```

```bash
#!/bin/bash
# build_android.sh - Script completo para compilar a múltiples arquitecturas

set -e  # Salir si hay error

# Configuración
KEYSTORE_PATH="release.keystore"
KEY_ALIAS="editor"
VERSION_NAME=$(grep '^version =' Cargo.toml | cut -d '"' -f2)
BUILD_DIR="target/android-build"

# Colores para output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

echo -e "${GREEN}=== Compilando Rust Editor para Android v${VERSION_NAME} ===${NC}"

# Crear directorio de build
mkdir -p $BUILD_DIR

# Función para compilar una arquitectura
compile_arch() {
    local arch=$1
    local target=$2
    
    echo -e "${YELLOW}Compilando para $arch ($target)...${NC}"
    
    # Añadir target si no existe
    rustup target add $target
    
    # Compilar en release
    cargo build --release --target $target
    
    # Copiar librería
    local lib_path="target/$target/release/librust_editor.so"
    local out_path="$BUILD_DIR/lib-$arch.so"
    
    if [ -f "$lib_path" ]; then
        cp "$lib_path" "$out_path"
        echo -e "${GREEN}✓ $arch compilado correctamente${NC}"
    else
        echo -e "${RED}✗ Error compilando $arch${NC}"
        exit 1
    fi
}

# Compilar arquitecturas soportadas
compile_arch "arm64-v8a" "aarch64-linux-android"
compile_arch "armeabi-v7a" "armv7-linux-androideabi"
# Opcional: descomentar si se necesita
# compile_arch "x86_64" "x86_64-linux-android"

echo -e "${GREEN}=== Creando APK ===${NC}"

# Crear estructura de APK
APK_DIR="$BUILD_DIR/apk"
mkdir -p $APK_DIR/lib/arm64-v8a
mkdir -p $APK_DIR/lib/armeabi-v7a

# Copiar librerías
cp $BUILD_DIR/lib-arm64-v8a.so $APK_DIR/lib/arm64-v8a/librust_editor.so
cp $BUILD_DIR/lib-armeabi-v7a.so $APK_DIR/lib/armeabi-v7a/librust_editor.so

# Crear AndroidManifest.xml
cat > $APK_DIR/AndroidManifest.xml << EOF
<?xml version="1.0" encoding="utf-8"?>
<manifest xmlns:android="http://schemas.android.com/apk/res/android"
    package="com.tueditor.app"
    android:versionCode="1"
    android:versionName="$VERSION_NAME">

    <!-- Permisos necesarios -->
    <uses-permission android:name="android.permission.WRITE_EXTERNAL_STORAGE" />
    <uses-permission android:name="android.permission.READ_EXTERNAL_STORAGE" />
    
    <!-- OpenGL ES 2.0+ para raylib -->
    <uses-feature android:glEsVersion="0x00020000" android:required="true" />
    
    <!-- Vulkan opcional para wgpu -->
    <uses-feature android:name="android.hardware.vulkan.level" android:required="false" />
    
    <!-- Soporte de texturas comprimidas -->
    <supports-gl-texture android:name="GL_EXT_texture_compression_etc2" />
    
    <application
        android:allowBackup="true"
        android:label="Rust Editor"
        android:theme="@android:style/Theme.NoTitleBar.Fullscreen"
        android:hasCode="false"
        android:extractNativeLibs="true">
        
        <activity
            android:name="android.app.NativeActivity"
            android:label="Rust Editor"
            android:configChanges="orientation|keyboardHidden|screenSize"
            android:exported="true">
            
            <meta-data
                android:name="android.app.lib_name"
                android:value="rust_editor" />
            
            <intent-filter>
                <action android:name="android.intent.action.MAIN" />
                <category android:name="android.intent.category.LAUNCHER" />
            </intent-filter>
        </activity>
    </application>
</manifest>
EOF

# Crear APK sin firmar
echo -e "${YELLOW}Empaquetando APK sin firmar...${NC}"
cd $APK_DIR
zip -r ../editor-unsigned.zip * > /dev/null
cd ../..

# Zipalign (optimizar)
echo -e "${YELLOW}Optimizando con zipalign...${NC}"
zipalign -f -p 4 $BUILD_DIR/editor-unsigned.zip $BUILD_DIR/editor-aligned.apk

# Firmar APK
echo -e "${YELLOW}Firmando APK con apksigner...${NC}"
apksigner sign \
    --ks $KEYSTORE_PATH \
    --ks-key-alias $KEY_ALIAS \
    --v1-signing-enabled true \
    --v2-signing-enabled true \
    --v3-signing-enabled true \
    --out $BUILD_DIR/rust-editor-v$VERSION_NAME.apk \
    $BUILD_DIR/editor-aligned.apk

# Verificar firma
echo -e "${YELLOW}Verificando firma...${NC}"
apksigner verify --verbose $BUILD_DIR/rust-editor-v$VERSION_NAME.apk

# Limpiar archivos temporales
rm $BUILD_DIR/editor-unsigned.zip $BUILD_DIR/editor-aligned.apk

echo -e "${GREEN}=== ¡COMPLETADO! ===${NC}"
echo -e "APK generado: ${YELLOW}$BUILD_DIR/rust-editor-v$VERSION_NAME.apk${NC}"
ls -lh $BUILD_DIR/rust-editor-v$VERSION_NAME.apk
```
